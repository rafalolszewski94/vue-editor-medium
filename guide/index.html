# Options

This library inherits all MediumEditor options (opens new window).

# Setting options

<template>
  <vue-editor-medium v-model="content" :options="options"/>
</template>

<script>
import { VueEditorMedium } from 'vue-editor-medium'

export default {
  components: {
    VueEditorMedium,
  },
  data() {
    return {
      content: 'This is my text',
      options: {
        static: true
      }
    }
  }
}
</script>
